#!/bin/bash

set -Eeuo pipefail

ORG="MON_ORGANIZATION"
BASE_DIR="/var/tmp/satellite_cv_ccv_export"
RUN_DATE="$(date +%F_%H%M%S)"
EXPORT_ROOT="${BASE_DIR}/run_${RUN_DATE}"
EXPORT_DIR="${EXPORT_ROOT}/exports"
REPORT_DIR="${EXPORT_ROOT}/reports"
LOG_FILE="${EXPORT_ROOT}/export.log"

mkdir -p "${EXPORT_DIR}" "${REPORT_DIR}"

exec > >(tee -a "${LOG_FILE}") 2>&1

echo "=============================================================="
echo " Satellite CV / CCV export - latest versions"
echo " Organization : ${ORG}"
echo " Export root  : ${EXPORT_ROOT}"
echo " Started at   : $(date)"
echo "=============================================================="

command -v hammer >/dev/null 2>&1 || {
  echo "ERROR: hammer command not found"
  exit 1
}

CV_REPORT="${REPORT_DIR}/cv_ccv_inventory.csv"
COMP_REPORT="${REPORT_DIR}/ccv_components.csv"
EXPORT_REPORT="${REPORT_DIR}/export_status.csv"

echo '"cv_id","name","label","type","description","latest_version","version_id","lifecycle_environments"' > "${CV_REPORT}"
echo '"ccv_name","ccv_id","component_cv_name","component_cv_id","component_latest_version"' > "${COMP_REPORT}"
echo '"name","type","latest_version","version_id","export_path","status","message"' > "${EXPORT_REPORT}"

sanitize_name() {
  echo "$1" | sed 's/[[:space:]]\+/_/g' | tr -cd '[:alnum:]_.-'
}

csv_escape() {
  local s="${1:-}"
  s="${s//\"/\"\"}"
  printf '"%s"' "$s"
}

detect_cv_type() {
  local cv_name="$1"
  local info
  info="$(hammer content-view info --organization "${ORG}" --name "${cv_name}" 2>/dev/null || true)"

  if echo "${info}" | grep -qiE '^Composite[[:space:]]*:[[:space:]]*yes'; then
    echo "CCV"
  elif echo "${info}" | grep -qiE '^Composite[[:space:]]*Content[[:space:]]*View'; then
    echo "CCV"
  else
    echo "CV"
  fi
}

extract_field() {
  local field="$1"
  local text="$2"
  echo "${text}" | awk -F': ' -v key="${field}" '$1 ~ "^"key"$" {sub($1": ",""); print; exit}'
}

echo
echo "Step 1 - Listing all Content Views"
mapfile -t CV_LINES < <(
  hammer --csv content-view list \
    --organization "${ORG}" \
    --fields "Id,Name,Label" | tail -n +2
)

if [ ${#CV_LINES[@]} -eq 0 ]; then
  echo "No Content Views found in ${ORG}"
  exit 0
fi

SUCCESS_COUNT=0
FAIL_COUNT=0
SKIP_COUNT=0

for line in "${CV_LINES[@]}"; do
  CV_ID="$(echo "${line}" | awk -F, '{print $1}' | xargs)"
  CV_NAME_RAW="$(echo "${line}" | awk -F, '{print $2}' | sed 's/^"//;s/"$//' | xargs)"
  CV_LABEL_RAW="$(echo "${line}" | awk -F, '{print $3}' | sed 's/^"//;s/"$//' | xargs)"

  echo
  echo "--------------------------------------------------------------"
  echo "Processing: ${CV_NAME_RAW} (ID=${CV_ID})"
  echo "--------------------------------------------------------------"

  CV_INFO="$(hammer content-view info \
    --organization "${ORG}" \
    --id "${CV_ID}" 2>/dev/null || true)"

  if [ -z "${CV_INFO}" ]; then
    echo "ERROR: unable to read info for ${CV_NAME_RAW}"
    echo "\"${CV_NAME_RAW}\",\"UNKNOWN\",\"\",\"\",\"FAILED\",\"Unable to read content-view info\"" >> "${EXPORT_REPORT}"
    FAIL_COUNT=$((FAIL_COUNT + 1))
    continue
  fi

  CV_TYPE="$(detect_cv_type "${CV_NAME_RAW}")"
  CV_DESC="$(extract_field "Description" "${CV_INFO}")"

  VERSION_LINE="$(hammer --csv content-view version list \
    --organization "${ORG}" \
    --content-view-id "${CV_ID}" \
    --fields "Id,Version,Environments" | tail -n +2 | sort -t, -k1,1nr | head -1 || true)"

  if [ -z "${VERSION_LINE}" ]; then
    echo "WARN: no published version found for ${CV_NAME_RAW}"
    printf '%s,%s,%s,%s,%s,%s,%s,%s\n' \
      "$(csv_escape "${CV_ID}")" \
      "$(csv_escape "${CV_NAME_RAW}")" \
      "$(csv_escape "${CV_LABEL_RAW}")" \
      "$(csv_escape "${CV_TYPE}")" \
      "$(csv_escape "${CV_DESC}")" \
      "$(csv_escape "")" \
      "$(csv_escape "")" \
      "$(csv_escape "")" >> "${CV_REPORT}"

    echo "\"${CV_NAME_RAW}\",\"${CV_TYPE}\",\"\",\"\",\"\",\"SKIPPED\",\"No published version found\"" >> "${EXPORT_REPORT}"
    SKIP_COUNT=$((SKIP_COUNT + 1))
    continue
  fi

  VERSION_ID="$(echo "${VERSION_LINE}" | awk -F, '{print $1}' | xargs)"
  VERSION_NAME="$(echo "${VERSION_LINE}" | awk -F, '{print $2}' | xargs)"
  ENVIRONMENTS="$(echo "${VERSION_LINE}" | cut -d, -f3- | sed 's/^"//;s/"$//')"

  printf '%s,%s,%s,%s,%s,%s,%s,%s\n' \
    "$(csv_escape "${CV_ID}")" \
    "$(csv_escape "${CV_NAME_RAW}")" \
    "$(csv_escape "${CV_LABEL_RAW}")" \
    "$(csv_escape "${CV_TYPE}")" \
    "$(csv_escape "${CV_DESC}")" \
    "$(csv_escape "${VERSION_NAME}")" \
    "$(csv_escape "${VERSION_ID}")" \
    "$(csv_escape "${ENVIRONMENTS}")" >> "${CV_REPORT}"

  if [ "${CV_TYPE}" = "CCV" ]; then
    echo "Collecting CCV components for ${CV_NAME_RAW}"

    COMPONENTS_CSV="$(hammer --csv content-view component list \
      --organization "${ORG}" \
      --content-view "${CV_NAME_RAW}" 2>/dev/null || true)"

    if [ -n "${COMPONENTS_CSV}" ]; then
      while IFS= read -r comp_line; do
        [ -z "${comp_line}" ] && continue
        echo "${comp_line}" | grep -q '^Content View ID,' && continue

        COMP_ID="$(echo "${comp_line}" | awk -F, '{print $1}' | xargs)"
        COMP_NAME="$(echo "${comp_line}" | awk -F, '{print $2}' | sed 's/^"//;s/"$//' | xargs)"
        COMP_LATEST="$(hammer --csv content-view version list \
          --organization "${ORG}" \
          --content-view-id "${COMP_ID}" \
          --fields "Version" 2>/dev/null | tail -n +2 | head -1 | xargs || true)"

        printf '%s,%s,%s,%s,%s\n' \
          "$(csv_escape "${CV_NAME_RAW}")" \
          "$(csv_escape "${CV_ID}")" \
          "$(csv_escape "${COMP_NAME}")" \
          "$(csv_escape "${COMP_ID}")" \
          "$(csv_escape "${COMP_LATEST}")" >> "${COMP_REPORT}"
      done <<< "${COMPONENTS_CSV}"
    fi
  fi

  SAFE_NAME="$(sanitize_name "${CV_NAME_RAW}")"
  DEST_DIR="${EXPORT_DIR}/${SAFE_NAME}_v${VERSION_NAME}"

  mkdir -p "${DEST_DIR}"

  echo "Type              : ${CV_TYPE}"
  echo "Latest version    : ${VERSION_NAME}"
  echo "Version ID        : ${VERSION_ID}"
  echo "Lifecycle envs    : ${ENVIRONMENTS}"
  echo "Destination       : ${DEST_DIR}"

  if hammer content-export complete version \
      --organization "${ORG}" \
      --id "${VERSION_ID}" \
      --destination-server localhost \
      --path "${DEST_DIR}"; then
    echo "OK: export done for ${CV_NAME_RAW}"
    echo "\"${CV_NAME_RAW}\",\"${CV_TYPE}\",\"${VERSION_NAME}\",\"${VERSION_ID}\",\"${DEST_DIR}\",\"SUCCESS\",\"Export completed\"" >> "${EXPORT_REPORT}"
    SUCCESS_COUNT=$((SUCCESS_COUNT + 1))
  else
    echo "ERROR: export failed for ${CV_NAME_RAW}"
    echo "\"${CV_NAME_RAW}\",\"${CV_TYPE}\",\"${VERSION_NAME}\",\"${VERSION_ID}\",\"${DEST_DIR}\",\"FAILED\",\"hammer content-export failed\"" >> "${EXPORT_REPORT}"
    FAIL_COUNT=$((FAIL_COUNT + 1))
  fi
done

echo
echo "=============================================================="
echo "Finished at : $(date)"
echo "Success     : ${SUCCESS_COUNT}"
echo "Failed      : ${FAIL_COUNT}"
echo "Skipped     : ${SKIP_COUNT}"
echo "Export root : ${EXPORT_ROOT}"
echo "Reports     : ${REPORT_DIR}"
echo "Log file    : ${LOG_FILE}"
echo "=============================================================="

echo
echo "Generated files:"
echo " - ${CV_REPORT}"
echo " - ${COMP_REPORT}"
echo " - ${EXPORT_REPORT}"
